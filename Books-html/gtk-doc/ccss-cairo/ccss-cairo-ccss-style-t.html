enum ccss_cairo_gap_side_t

typedef enum {
	CCSS_CAIRO_GAP_SIDE_LEFT,
	CCSS_CAIRO_GAP_SIDE_RIGHT,
	CCSS_CAIRO_GAP_SIDE_TOP,
	CCSS_CAIRO_GAP_SIDE_BOTTOM
} ccss_cairo_gap_side_t;

This enum mirrors GtkPositionType.

CCSS_CAIRO_GAP_SIDE_LEFT

 gap on the right.

CCSS_CAIRO_GAP_SIDE_TOP
